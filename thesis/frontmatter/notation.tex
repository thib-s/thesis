\addcontentsline{toc}{chapter}{Acronyms and Notation}

\section*{Acronyms}
\begin{description}
    \item[AOC] Adaptive Orthogonal Convolutions
    \item[BCOP] Block Convolution Orthogonal Parameterization
    \item[CNN] Convolutional Neural Network
    \item[RKO] Reshaped Kernel Orthogonalization
    \item[SDP] Semi-Definite Programming
    \item[SLL] SDP-based Lipschitz Layers
    \item[SOC] Skew Orthogonal Convolution
\end{description}

\section*{Notation}
\begin{description}
    \item[$\R$] The set of real numbers
    \item[$\Kbf$] Convolution kernel
    \item[$\KbfS$] The space of $c_o \times c_i \times k \times k$ convolutions
    \item[$\mathbconv$] Block-convolution operator
    \item[$\toep{K}$] Toeplitz matrix associated with kernel $\Kbf$
    \item[$\toepstride{s}$] Stride masking matrix
    \item[$\convker{s}$] Convolution operator with stride $s$
\end{description}
